82       public String pathBuilder() {
83           String separator = System.getProperty( "file.separator" );
84           String home = null;
85   
86           //Make Program Compatible With Windows (I use windows :P )
87           if (System.getProperty( "os.name" ).startsWith( "Windows" )) {
88               home = System.getProperty( "user.dir" );
89           } else if (System.getProperty( "os.name" ).startsWith( "Windows 10" )) {
90               home = System.getProperty( "user.dir" );
91           } else {
92               home = System.getProperty( "user.home" );
93           }
94           String path = home + separator + "CS1Files" + separator + "data" + separator;
95           return path;
96       }

98       public BufferedImage randomlyGetImageFromPath(String path) throws IOException {
99   
100          Random rgen = new Random();
101          File dataFolder = new File( path );
102  
103          // Randomly choose any image from the folder (Scan for Jpg and PNG);
104          FilenameFilter filter = (dir, name) -> {
105              String lowercaseName = name.toLowerCase();
106              if (lowercaseName.endsWith( ".jpg" )) {
107                  return true;
108              } else if (lowercaseName.endsWith( ".png" )) {
109                  return true;
110              } else if (lowercaseName.endsWith( ".jpeg" )) {
111                  return true;
112              } else {
113                  return false;
114              }
115          };
116          File[] listOfFiles = dataFolder.listFiles( filter );
117          File randomPicture = listOfFiles[rgen.nextInt( listOfFiles.length )];
118          return ImageIO.read( randomPicture );
119      }

121      public Color getPixelColor(int x, int y) {
122          int rgb = image.getRGB( x, y );
123          int red = (rgb & 0x00ff0000) >> 16;
124          int green = (rgb & 0x0000ff00) >> 8;
125          int blue = (rgb & 0x000000ff);
126          Color PixelColor = new Color( red, green, blue );
127          System.out.println( "Thread #:" + Thread.currentThread().getId() + "|" + PixelColor.toString() );
128          return PixelColor;
129      }

131      public void paintOnePixel(int workingPixel, int workingRow) {
132          Color pixelColor = getPixelColor( workingPixel, workingRow );
133          painter.setColor( pixelColor );
134          //Changing this may make pictures generate faster
135          painter.mrt( 1 );
136          painter.paint( rectanglePixel );
137      }

139      public void paintOneRow(int workingRow) {
140          int pixelIterator = 0;
141          while (pixelIterator < image.getWidth()) {
142              paintOnePixel( pixelIterator, workingRow );
143              pixelIterator = pixelIterator + 1;
144              System.out.println( "Thread #:" + Thread.currentThread().getId() + "| DEBUG: Row Pixel Count:  " + pixelIterator );
145          }
146      }

148      public void paintTopRows() {
149          int rowCount = 0;
150          // Starts Off From Corner of Canvas
151          painter.moveTo( new Point2D.Double( 0, 0 ) );
152          for (int rowIterator = 0; rowIterator < image.getHeight() / 2; rowIterator++) {
153              paintOneRow( rowIterator );
154              rowCount = rowCount + 1;
155              System.out.println( "Thread #:" + Thread.currentThread().getId() + "| DEBUG: Row Count:  " + rowCount );
156              // Makes a new Row
157              painter.moveTo( new Point2D.Double( 0, rowIterator ) );
158              System.out.println( "-------------------------------------------------" );
159              System.out.println( "Thread #: " + Thread.currentThread().getId() + "| DEBUG: NEXT ROW" );
160              System.out.println( "-------------------------------------------------" );
161              if (rowCount == image.getHeight()) {
162                  System.out.println( "-------------------------------------------------" );
163                  System.out.println( "Thread #: " + Thread.currentThread().getId() + "| Picture Finished Drawing" );
164                  System.out.println( "-------------------------------------------------" );
165                  break;
166              }
167  
168  
169          }
170      }
171      public void paintOneBottomRow(int workingRow) {
172          int pixelIterator = 0;
173          while (pixelIterator < image.getWidth()) {
174              paintOnePixel( pixelIterator, workingRow );
175              pixelIterator = pixelIterator + 1;
176              System.out.println( "Thread #:" + Thread.currentThread().getId() + "| DEBUG: Row Pixel Count:  " + pixelIterator );
177          }
178      }

180      public void paintBottomRows() {
181          System.out.println("Paint Bottom Rows");
182          int rowCount = 0;
183          // Starts Off at Bottom Corner
184          painter.moveTo( new Point2D.Double( 0, height ) );
185          int rowIterator = -1+height; // -1 to make getPixelColor Happy
186          int rowMover = 3; // 3 offset to compensate for gap and over draw
187          while( rowIterator > image.getHeight() / 2) {
188              paintOneBottomRow( rowIterator );
189              rowIterator = rowIterator - 1;
190              rowMover = rowMover + 1;
191              rowCount = rowCount + 1;
192              System.out.println( "Thread #:" + Thread.currentThread().getId() + "| DEBUG: Row Count:  " + rowCount );
193              // Makes a new Row
194              painter.moveTo( new Point2D.Double( 0, height - rowMover));
195              System.out.println( "-------------------------------------------------" );
196              System.out.println( "Thread #: " + Thread.currentThread().getId() + "| DEBUG: NEXT ROW" );
197              System.out.println( "-------------------------------------------------" );
198              if (rowCount == image.getHeight()) {
199                  System.out.println( "-------------------------------------------------" );
200                  System.out.println( "Thread #: " + Thread.currentThread().getId() + "| Picture Finished Drawing" );
201                  System.out.println( "-------------------------------------------------" );
202                  break;
203              }
204  
205  
206          }
