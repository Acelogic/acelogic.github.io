5     * - order, the number of faces 
6     * - top, the value of the top face

8    public class Die {

11       private int top;
12   
13       // THE CONSTRUCTORS
14       public Die() {

18

21           top = (int) ((Math.random() * nrOfSides) + 1);

24       // THE METHODS (BEHAVIOR)
25       public int top() {

27       }
28   
29       public void roll() {
