3     * sequences classified as "Basic" sequences in the Modular Melody World

6    package mmw;
7    import composer.SComposer;
8    public class BasicsListener {
9        public static void main(String[] args) {
10           SComposer c = new SComposer();
11           c.text();
12           System.out.println("c.mms1 ..."); c.mms1(); space(c);
13           System.out.println("c.mms2 ..."); c.mms2(); space(c);
14           System.out.println("c.mms3 ..."); c.mms3(); space(c);
15           System.out.println("c.mms4 ..."); c.mms4(); space(c);
16           System.out.println("c.mms5 ..."); c.mms5(); space(c);
17           System.out.println("c.mms6 ..."); c.mms6(); space(c);
18           System.out.println("c.mms7 ..."); c.mms7(); space(c);
19           System.out.println("c.mms8 ..."); c.mms8(); space(c);
20           c.untext();
21       }
22       private static void space(SComposer c) {
23           c.untext(); c.rest(2); c.text();
24       }
25   }
